inicio Calculadora
    variaveis: valor1, valor2
    ler(valor1, valor2)

    escreva("A soma de ", valor1, "+", valor2," = ", valor1+valor2)

    escreva("A subitração de ", valor1, "-", valor2," = ", valor1-valor2)
